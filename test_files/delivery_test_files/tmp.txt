program incaseTest

    declare x,i,y; 

    #main#
    {
        i := 0;
        input(x);

        incase 
            case(x > 10)
            {
                x := x + 5;
                input(y);
            }
            case(y < x)
            {
                y := y * 10;
            }
            case (y = x)
            {
                forcase 
                    case (y<10)
                    {
                        y := 10;
                    }
                    case (y > 10)
                    {
                        y := y - 10;
                    }
                    default
                    {
                        print(-1000)
                    }
            
            }
            case ( [y = 10] or [y = 0])
            {
                print(y);
            }
        ;
        print(x);
    }.
